import torch
import torch.nn as nn
from transformers import AutoModel


class MiniLMWithPostTransformer(nn.Module):
    def __init__(
        self,
        model_name: str,
        num_labels: int,
        hidden_size: int = 384,
        num_heads: int = 6,
        dropout: float = 0.1
    ):
        super().__init__()

        # Backbone: MiniLM
        self.encoder = AutoModel.from_pretrained(model_name)

        # Post-MiniLM Transformer block (1 lightweight layer)
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=hidden_size,
            nhead=num_heads,
            dim_feedforward=hidden_size * 4,
            dropout=dropout,
            activation="gelu",
            batch_first=True,
            norm_first=True
        )

        self.post_transformer = nn.TransformerEncoder(
            encoder_layer,
            num_layers=1
        )

        # Classification head
        self.classifier = nn.Sequential(
            nn.Linear(hidden_size, 256),
            nn.GELU(),
            nn.LayerNorm(256),
            nn.Dropout(0.2),
            nn.Linear(256, num_labels)
        )

    def forward(self, input_ids):
        # MiniLM forward
        outputs = self.encoder(input_ids=input_ids)
        hidden_states = outputs.last_hidden_state  # [B, T, 384]

        # Post transformer
        hidden_states = self.post_transformer(hidden_states)

        # Mean pooling (no attention mask)
        pooled = hidden_states.mean(dim=1)  # [B, 384]

        # Classifier
        logits = self.classifier(pooled)
        return logits
