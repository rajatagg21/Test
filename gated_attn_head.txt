import torch
import torch.nn as nn


class BottleneckTransformer(nn.Module):
    def __init__(
        self,
        hidden_dim,
        attn_dim,
        num_heads,
        num_layers,
        dropout
    ):
        super().__init__()

        self.num_layers = num_layers
        if num_layers == 0:
            return

        self.down = nn.Linear(hidden_dim, attn_dim, bias=False)

        self.attn_layers = nn.ModuleList([
            nn.MultiheadAttention(
                embed_dim=attn_dim,
                num_heads=num_heads,
                dropout=dropout,
                batch_first=True
            )
            for _ in range(num_layers)
        ])

        self.up = nn.Linear(attn_dim, hidden_dim, bias=False)
        self.gate = nn.Linear(hidden_dim, 1)
        self.ln = nn.LayerNorm(hidden_dim)

    def forward(self, x):
        if self.num_layers == 0:
            return x

        residual = x
        x = self.down(x)

        for attn in self.attn_layers:
            x, _ = attn(x, x, x, need_weights=False)

        x = self.up(x)
        gate = torch.sigmoid(self.gate(residual))

        return self.ln(residual + gate * x)



class AdsClassifier(nn.Module):
    def __init__(self, hidden_size, num_labels):
        super().__init__()

        # ---- INTERNAL ARCH CONTROL ----
        attn_dim = hidden_size // 4     # 384 â†’ 96
        num_heads = 4
        num_attn_layers = 1
        dropout = 0.1
        mlp_dim = 256
        # --------------------------------

        self.transformer = BottleneckTransformer(
            hidden_dim=hidden_size,
            attn_dim=attn_dim,
            num_heads=num_heads,
            num_layers=num_attn_layers,
            dropout=dropout
        )

        self.mlp = nn.Sequential(
            nn.Linear(hidden_size, mlp_dim),
            nn.LayerNorm(mlp_dim),
            nn.GELU(),
            nn.Dropout(dropout),
            nn.Linear(mlp_dim, num_labels)
        )

    def forward(self, hidden_states):
        x = self.transformer(hidden_states)
        x = x.mean(dim=1)  # mask-free mean pooling
        return self.mlp(x)



class MiniAdsModel(nn.Module):
    def __init__(self, encoder, hidden_size, num_labels):
        super().__init__()

        self.encoder = encoder
        self.classifier = AdsClassifier(
            hidden_size=hidden_size,
            num_labels=num_labels
        )

    def forward(self, input_ids, attention_mask=None):
        outputs = self.encoder(
            input_ids=input_ids,
            attention_mask=attention_mask
        )
        return self.classifier(outputs.last_hidden_state)
