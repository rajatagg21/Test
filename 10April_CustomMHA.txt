import tensorflow as tf
from tensorflow.keras import layers

class DistillableMHA(layers.Layer):
    def __init__(self, num_heads, embed_dim, **kwargs):
        super(DistillableMHA, self).__init__(**kwargs)
        self.num_heads = num_heads
        self.embed_dim = embed_dim
        
        # Internal Head Dimension Calculation
        if embed_dim % num_heads != 0:
            raise ValueError(
                f"embed_dim ({embed_dim}) must be divisible by "
                f"num_heads ({num_heads})."
            )
        
        self.head_dim = embed_dim // num_heads
        
        # Projections: Output matches embed_dim
        self.q_dense = layers.Dense(embed_dim)
        self.k_dense = layers.Dense(embed_dim)
        self.v_dense = layers.Dense(embed_dim)
        
        # Final output projection (W_O)
        self.out_dense = layers.Dense(embed_dim)

    def split_heads(self, x, batch_size):
        # Reshape: (batch, seq_len, num_heads, head_dim)
        x = tf.reshape(x, (batch_size, -1, self.num_heads, self.head_dim))
        # Transpose to: (batch, num_heads, seq_len, head_dim)
        return tf.transpose(x, perm=[0, 2, 1, 3])

    def call(self, query, value, key=None, mask=None, training=None):
        if key is None:
            key = value
        
        batch_size = tf.shape(query)[0]

        # 1. Linear Projections & Head Splitting
        q = self.split_heads(self.q_dense(query), batch_size)
        k = self.split_heads(self.k_dense(key), batch_size)
        v = self.split_heads(self.v_dense(value), batch_size)

        # 2. Q-K Attention Distribution (Rel_QK)
        # Result: (batch, num_heads, seq_len_q, seq_len_k)
        qk_logits = tf.matmul(q, k, transpose_b=True)
        qk_logits = qk_logits / tf.math.sqrt(tf.cast(self.head_dim, tf.float32))
        
        if mask is not None:
            # Apply additive mask (usually -1e9 for padding/causal)
            qk_logits += mask
            
        attn_weights = tf.nn.softmax(qk_logits, axis=-1)

        # 3. Value-Value Relation (Rel_VV) 
        # Result: (batch, num_heads, seq_len_v, seq_len_v)
        vv_logits = tf.matmul(v, v, transpose_b=True)
        vv_logits = vv_logits / tf.math.sqrt(tf.cast(self.head_dim, tf.float32))
        
        # Note: Usually VV relations in MiniLM use the same mask as QK 
        # to ignore padding tokens in the relation matrix.
        if mask is not None:
            vv_logits += mask
            
        vv_relation = tf.nn.softmax(vv_logits, axis=-1)

        # 4. Concatenate and Project Output
        # (batch, num_heads, seq_len_q, head_dim)
        weighted_v = tf.matmul(attn_weights, v)
        weighted_v = tf.transpose(weighted_v, perm=[0, 2, 1, 3])
        
        # Merge heads back to embed_dim
        concat_v = tf.reshape(weighted_v, (batch_size, -1, self.embed_dim))
        output = self.out_dense(concat_v)

        return output, attn_weights, vv_relation
