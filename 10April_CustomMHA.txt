import tensorflow as tf
from tensorflow.keras import layers

class DistillableMHA(layers.Layer):
    def __init__(self, num_heads, key_dim, **kwargs):
        super(DistillableMHA, self).__init__(**kwargs)
        self.num_heads = num_heads
        self.key_dim = key_dim
        
        # Projections for Q, K, and V
        self.q_dense = layers.Dense(num_heads * key_dim)
        self.k_dense = layers.Dense(num_heads * key_dim)
        self.v_dense = layers.Dense(num_heads * key_dim)
        
        # Final output projection
        self.out_dense = layers.Dense(kwargs.get('output_dim', num_heads * key_dim))

    def split_heads(self, x, batch_size):
        # Reshape to (batch, seq_len, num_heads, key_dim)
        x = tf.reshape(x, (batch_size, -1, self.num_heads, self.key_dim))
        # Transpose to (batch, num_heads, seq_len, key_dim)
        return tf.transpose(x, perm=[0, 2, 1, 3])

    def call(self, query, value, key=None, training=None):
        if key is None:
            key = value
        
        batch_size = tf.shape(query)[0]

        # 1. Linear Projections
        q = self.split_heads(self.q_dense(query), batch_size)
        k = self.split_heads(self.k_dense(key), batch_size)
        v = self.split_heads(self.v_dense(value), batch_size)

        # 2. Q-K Attention Distribution (Rel_QK)
        # Shape: (batch, num_heads, seq_len_q, seq_len_k)
        qk_logits = tf.matmul(q, k, transpose_b=True)
        qk_logits = qk_logits / tf.math.sqrt(tf.cast(self.key_dim, tf.float32))
        attn_weights = tf.nn.softmax(qk_logits, axis=-1)

        # 3. Value-Value Relation (Rel_VV) - The MiniLM specific part
        # Shape: (batch, num_heads, seq_len_v, seq_len_v)
        vv_logits = tf.matmul(v, v, transpose_b=True)
        vv_logits = vv_logits / tf.math.sqrt(tf.cast(self.key_dim, tf.float32))
        vv_relation = tf.nn.softmax(vv_logits, axis=-1)

        # 4. Standard MHA Output calculation
        # weighted_v shape: (batch, num_heads, seq_len_q, key_dim)
        weighted_v = tf.matmul(attn_weights, v)
        weighted_v = tf.transpose(weighted_v, perm=[0, 2, 1, 3])
        concat_v = tf.reshape(weighted_v, (batch_size, -1, self.num_heads * self.key_dim))
        output = self.out_dense(concat_v)

        # Return everything needed for training and distillation
        return output, attn_weights, vv_relation

# Example Usage:
